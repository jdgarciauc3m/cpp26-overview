\subsection{Introduction to parallel algorithms}

\begin{frame}[t,fragile]{Finding in parallel}
\begin{itemize}

\item Finding with a predicate.

\begin{lstlisting}
std::vector<double> v = get_values();
auto it = std::find_if(v.begin(), v.end(), [](double x) { return x>0; });
\end{lstlisting}

\mode<presentation>{\vfill\pause}
\item Finding in parallel with an execution policy and iterators (C++17)

\begin{lstlisting}
std::vector<double> v = get_values();
auto it = std::find_if(std::execution::par, v.begin(), v.end(), [](double x) { return x>0; });
\end{lstlisting}

\mode<presentation>{\vfill\pause}
\item Finding in parallel with an execution policy and ranges (C++26).
\begin{lstlisting}
std::vector<double> v = get_values();
auto it = std::ranges::find_if(std::execution::par, v, [](double x) { return x>0; });
\end{lstlisting}

\end{itemize}
\end{frame}

\begin{frame}[t,fragile]{Combining algorithms}
\begin{itemize}

    \item Finding the last element whose square is greater than 100 (C++17).
\begin{lstlisting}
std::vector<double> v = get_values();
std::transform(std::execution::par, v.begin(), v.end(), v.begin(), [](double x) { return x*x; });
std::reverse(std::execution::par, v.begin(), v.end());
auto it = std::find_if(std::execution::par, v.begin(), v.end(), [](double x) { return x>100; });
\end{lstlisting}

    \mode<presentation>{\vfill\pause}
    \item \textbad{Drawbacks}:
    \begin{itemize}
        \item Three separate algorithms, each with its own overhead.
        \item Temporary computations might be skipped.
    \end{itemize}

    \mode<presentation>{\vfill\pause}
    \item Combining ranges and parallel execution policies (C++20).
    
\begin{lstlisting}
std::vector<double> v = get_values();
auto compute = v 
             | std::views::transform([](double x) { return x*x; })
             | std::views::reverse;
auto it = std::find_if(std::execution::par, 
    std::begin(compute), std::end(compute), 
    [](double x) { return x>100; });
\end{lstlisting}

\end{itemize}
\end{frame}

\begin{frame}[t,fragile]{Combining algorithms with ranges}
\begin{itemize}

    \item Combining ranges and parallel execution policies (C++26).
\begin{lstlisting}
std::vector<double> v = get_values();
auto compute = v 
             | std::views::transform([](double x) { return x*x; })
             | std::views::reverse;
auto it = std::ranges::find_if(std::execution::par, 
    compute, 
    [](double x) { return x>100; });
\end{lstlisting}

     \item \textmark{Warning}: The temporary variable \cppcode{compute} is necessary to avoid a dangling reference. 
\begin{lstlisting}
std::vector<double> v = get_values();
auto it = std::ranges::find_if(std::execution::par, 
    @v@
    @| std::views::transform([](double x) { return x*x; })@
    @| std::views::reverse@,
    [](double x) { return x>100; });
\end{lstlisting}

\end{itemize}
\end{frame}
